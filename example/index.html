下拉刷新 已注册在该页面

requestHybrid({
  action: 'registerRefresh',
  // 下拉后执行的回调, 就全部刷新
  callback: function(data) {
    requestHybrid({ action: 'hideRefresh' })
    location.reload();  // 前端直接刷新，或者发起请求更新
  }
})
